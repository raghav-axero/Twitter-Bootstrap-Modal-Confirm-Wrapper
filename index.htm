Twitter Bootstrap confirm modal

A simple wrapper for confirm modal based Twitter Bootstrap modal.

Learn more »

Basic example

    $('#ConfirmButton1').on('click', function(e) {
        $.confirm('This is a simple confirm message.');
    });



Add custom title

    $('#ConfirmButton2').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title:'Confirm dialog'
        });
    });



Modify button's text

    $('#ConfirmButton3').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes, I want to',
            cancelButtonText: 'No, I am not'
        });
    });



Add button's callbacks

    $('#ConfirmButton4').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes, I want to',
            cancelButtonText: 'No, I am not',
            onConfirm:function() {
                console.log('You selected confirm button');
            },
            onCancel:function() {
                console.log('You selected cancel button');        
            },                    
        });
    });

Add custom class to buttons

    $('#ConfirmButton5').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            confirmButtonClass: 'btn-info',
            cancelButtonClass: 'btn-danger'
        });
    });

Hide header close button

    $('#ConfirmButton6').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            showHeaderCloseButton: false
        });
    });

Hide header

    $('#ConfirmButton7').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            showHeader: false
        });
    });

Show icons on buttons

    $('#ConfirmButton8').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            confirmButtonIcon:'',
            cancelButtonIcon: ''
        });
    });

Reverse buttons order

    $('#ConfirmButton9').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            reverseButtonsOrder:true
        });
    });

Close modal from with in button's callback 

    $('#ConfirmButton10').on('click', function (e) {
        $.confirm('Are you sure you want to proceed?', {
            title: 'Confirm dialog',
            confirmButtonText: 'Yes',
            cancelButtonText: 'No',
            hideOnConfirm: false,
            onConfirm: function () {
                var modal = $(this);
                $.get('index.htm', function (data) {
                    modal.modal('hide');
                });
            }
        });
    });
